\begin{frame}{Alpha Estimation: Carhart Four-Factor Model}
\begin{columns}[T]
\column{0.54\textwidth}
\begin{block}{The model (Carhart, 1997)}
{\footnotesize For each fund $i$ with $\geq 24$ monthly observations:}
\begin{align*}
R_{i,t} - RF_t = \,&\alpha_i + \beta_{1,i}\,\mathrm{MKT}_t + \beta_{2,i}\,\mathrm{SMB}_t \\
                   &+ \beta_{3,i}\,\mathrm{HML}_t + \beta_{4,i}\,\mathrm{MOM}_t + \varepsilon_{i,t}
\end{align*}
{\footnotesize
\begin{itemize}
  \item \textbf{MKT:} excess market return $(R_M - RF)$; \textbf{SMB/HML:} size \& value (Fama--French, 1993); \textbf{MOM:} momentum (Jegadeesh--Titman, 1993)
  \item Factors from Kenneth French Data Library (monthly)
  \item Estimated \emph{independently} per fund via OLS; $\hat{\alpha}_i$ annualised $\times 12$
  \item Net alpha deducts one-twelfth of the yearly expense ratio 
\end{itemize}}
\end{block}

\column{0.43\textwidth}
\begin{block}{Specification rationale}
{\footnotesize
\begin{itemize}
  \item \textbf{Carhart over FF3:} momentum absorbs short-run persistence; omitting it spuriously assigns momentum-tilted funds genuine alpha
  \item \textbf{NW HAC SEs} (Newey \& West, 1987), \textbf{6-month Bartlett lag:} fund residuals exhibit positive serial autocorrelation; OLS SEs are downward-biased. $t(\hat{\alpha}_i)$ statistics are the direct inputs to the bootstrap — applied \textbf{symmetrically} in both actual and simulated distributions
  \item \textbf{Gross returns:} static ER approximation introduces measurement error in residuals; gross/net distinction handled at reporting stage
  \item \textbf{Min.\ 24 obs:} ensures reliable estimates; bootstrap uses same threshold
\end{itemize}}
\end{block}
\end{columns}
\end{frame}

\begin{frame}{Bootstrap Methodology: Why and What}
\begin{columns}[T]
\column{0.50\textwidth}
\begin{block}{Two problems with fund-by-fund $t$-tests}
\begin{enumerate}
  \item \textbf{Multiple comparisons:} with $N \approx 1{,}900$ funds tested simultaneously, non-trivial significance by chance alone (false discoveries)
  \item \textbf{Non-normality:} fund residuals are non-normal; standard $t$-test distributional assumptions fail
\end{enumerate}
\end{block}

\begin{block}{Fama \& French (2010) solution}
Build an \emph{empirical} null distribution from the data itself:
{\footnotesize
\begin{itemize}
  \item Null: $H_0\!: \alpha_i = 0\ \forall\, i$
  \item Impose the null on each fund's return series
  \item Resample calendar months (preserving cross-sectional factor correlation)
  \item Compare actual $t$-statistic distribution against the simulated null
\end{itemize}}
\end{block}

\column{0.47\textwidth}
\begin{block}{Complementary tools}
{\footnotesize \textbf{Storey (2002) $\hat{\pi}_0$:} proportion of true zero-alpha funds. Applied to mutual funds by Barras, Scaillet \& Wermers (BSW, 2010).

\smallskip
\textbf{BSW four-way decomposition:} given $\hat{\pi}_0$, separate the observed tails into genuinely \red{unskilled} ($T^-_\gamma$), genuinely \green{skilled} ($T^+_\gamma$), and lucky/unlucky (false discoveries, $F_\gamma$).

\smallskip
Together these answer: \emph{how many funds are genuinely skilled vs.\ merely lucky?}}
\end{block}
\end{columns}
\end{frame}

% ══════════════════════════════════════════════════════════
%  SLIDE 13 — BOOTSTRAP PROCEDURE
% ══════════════════════════════════════════════════════════
\begin{frame}{Bootstrap Procedure (FF 2010 + NW HAC Extension)}
\begin{enumerate}
  \setlength\itemsep{4pt}
  \item \textbf{Alpha demeaning} — Impose the zero-alpha null on each fund's return series:
  $\tilde{y}_{i,t} = (R_{i,t} - RF_t) - \hat{\alpha}_i$

  \item \textbf{Calendar-month index mapping} — $T=348$ global months. Each fund's observations are mapped to their global position; this preserves \emph{cross-sectional factor return dependence} when resampling.

  \item \textbf{Bootstrap resampling} — In each of $B=10{,}000$ iterations, draw $T$ months with replacement from the global index. For each fund, retain only its own observed months in the draw.

  \item \textbf{Simulated alpha estimation} — Re-estimate the Carhart model on resampled zero-alpha returns. Divide by \textbf{Newey-West HAC SEs} (same 6-month lag as actual regressions).

  \item \textbf{Percentile extraction} — Compute cross-sectional percentiles of simulated $t$-statistics per iteration; report \emph{mean simulated percentile} across all $B$ iterations alongside the \emph{actual percentile}.
\end{enumerate}
\end{frame}

\begin{frame}{Storey (2002) $\hat{\pi}_0$ and BSW Four-Way Decomposition}
\begin{columns}[T]
\column{0.50\textwidth}
\begin{block}{Storey (2002) estimator}
$p$-values for truly zero-alpha funds are $\mathrm{Uniform}(0,1)$. For $\lambda = 0.5$:
\[
\hat{\pi}_0 = \frac{|\{p_i > \lambda\}|}{N\,(1-\lambda)}
\]
\begin{itemize}
  \item Funds with $p_i > 0.5$ provide no evidence against $H_0$
  \item Estimator is \textbf{conservative} — overestimates $\pi_0$ when skilled funds exist
  \item Population: $N=1{,}899$ active funds, two-sided NW $p$-values from Carhart alphas
\end{itemize}
\end{block}

\column{0.47\textwidth}

\begin{frame}{What's Next: Testing Behavioral Hypotheses H1--H4}
\begin{columns}[T]
\column{0.48\textwidth}
{\small
\begin{block}{H1 — Sentiment-Convexity}
\textit{Does investor sentiment amplify the convex flow-performance relationship?}\\[3pt]
\textbf{Framework:} Sirri \& Tufano (1998) piecewise flow regression $\times$ Baker--Wurgler (2006) sentiment index.
\end{block}

\vspace{3pt}

\begin{block}{H2 — Illusion of Control}
\textit{Do investors experiencing the illusion of control allocate more to active funds?}\\[3pt]
\textbf{Framework:} Margin debt as leverage-appetite proxy $\times$ flow--performance interaction; motivated by Langer (1975) and Grinblatt \& Keloharju (2009).
\end{block}
}

\column{0.48\textwidth}
{\small
\begin{block}{H3 — Lottery Demand}
\textit{Do investors treat active funds as delegated lottery tickets?}\\[3pt]
\textbf{Framework:} Carhart $R^2$ as activeness proxy (Amihud \& Goyenko, 2013); flow sensitivity to idiosyncratic volatility and skewness; inspired by Kumar (2009).
\end{block}

\vspace{3pt}

\begin{block}{H4 — Fee Elasticity / Psychological Premium}
\textit{Is there a measurable \emph{shadow price} for the utility of active management?}\\[3pt]
\textbf{Framework:} Shadow price via implied MRS from fee-flow elasticity — \textbf{the unifying structural contribution} of the dissertation (Statman, 2004; Berk \& Green, 2004).
\end{block}
}

\end{columns}
\end{frame}

% ══════════════════════════════════════════════════════════
%  H1 — SENTIMENT-CONVEXITY
% ══════════════════════════════════════════════════════════
\begin{frame}{H1 — Sentiment-Convexity: Model}
\begin{columns}[T]
\column{0.54\textwidth}
\begin{block}{The model (Sirri \& Tufano, 1998 + Baker \& Wurgler, 2006)}
{\footnotesize Piecewise flow regression with sentiment interaction:}
{\footnotesize
\begin{align*}
\mathrm{Flow}_{i,t} = \;&\alpha_i + \beta_1 R^{\mathrm{LOW}}_{i,t-1} + \beta_2 R^{\mathrm{MID}}_{i,t-1} + \beta_3 R^{\mathrm{HIGH}}_{i,t-1} \\
&+ \delta_1 (R^{\mathrm{LOW}}_{i,t-1} \times D^{\mathrm{SENT}}_t) \\
&+ \delta_2 (R^{\mathrm{MID}}_{i,t-1} \times D^{\mathrm{SENT}}_t) \\
&+ \delta_3 (R^{\mathrm{HIGH}}_{i,t-1} \times D^{\mathrm{SENT}}_t) \\
&+ \lambda D^{\mathrm{SENT}}_t + \gamma \mathbf{X}_{i,t-1} + \varepsilon_{i,t}
\end{align*}}
{\footnotesize
\begin{itemize}
  \item $R^{\mathrm{LOW/MID/HIGH}}$: fractional rank segments (bottom 20\%, middle 60\%, top 20\%)
  \item $D^{\mathrm{SENT}}_t$: high-sentiment dummy, Baker--Wurgler (2006) orthogonalised index
  \item $\mathbf{X}_{i,t-1}$: log TNA, fund age, expense ratio, lagged flow
  \item Fund and time FE; NW HAC standard errors
\end{itemize}}
\end{block}

\column{0.43\textwidth}
\begin{block}{Identification}
{\footnotesize
\begin{itemize}
  \item \textbf{Hypothesis:} $\delta_3 > 0$ — high sentiment steepens the top-performance slope (winner chasing amplified)
  \item \textbf{Auxiliary:} $\delta_1 \geq 0$ — sentiment does \emph{not} amplify outflows from losers
  \item \textbf{Asymmetry test:} $\delta_3 > \delta_1$ distinguishes sentiment-driven convexity from uniform sentiment lift
  \item \textbf{Mechanism:} Representativeness heuristic (Kahneman \& Tversky, 1974) — investors extrapolate past returns most aggressively when optimistic
\end{itemize}}
\end{block}

\begin{block}{Novelty}
{\footnotesize Sirri \& Tufano (1998) establish \emph{static} convexity; this study conditions it on sentiment, decomposing the effect separately across the performance distribution.}
\end{block}

\end{columns}
\end{frame}

% ══════════════════════════════════════════════════════════
%  H2 — DISPOSITION-CONTROL
% ══════════════════════════════════════════════════════════
\begin{frame}{H2 — Disposition-Control: Model}
\begin{columns}[T]
\column{0.54\textwidth}
\begin{block}{The model (Langer, 1975; Shefrin \& Statman, 1985)}
{\footnotesize Same piecewise base; regime variable replaced by margin debt growth:}
{\footnotesize
\begin{align*}
\mathrm{Flow}_{i,t} = \;&\alpha_i + \beta_1 R^{\mathrm{LOW}}_{i,t-1} + \beta_2 R^{\mathrm{MID}}_{i,t-1} + \beta_3 R^{\mathrm{HIGH}}_{i,t-1} \\
&+ \delta^{\mathrm{MD}}_1 (R^{\mathrm{LOW}}_{i,t-1} \times \Delta\mathrm{MD}_t) \\
&+ \delta^{\mathrm{MD}}_2 (R^{\mathrm{MID}}_{i,t-1} \times \Delta\mathrm{MD}_t) \\
&+ \delta^{\mathrm{MD}}_3 (R^{\mathrm{HIGH}}_{i,t-1} \times \Delta\mathrm{MD}_t) \\
&+ \lambda \Delta\mathrm{MD}_t + \gamma \mathbf{X}_{i,t-1} + \varepsilon_{i,t}
\end{align*}}
{\footnotesize
\begin{itemize}
  \item $\Delta\mathrm{MD}_t$: YoY growth in FINRA aggregate margin debt (monthly, from 1997)
  \item Robustness: CBOE call-to-put ratio as second proxy for control illusion
  \item Key coefficient: $\delta^{\mathrm{MD}}_1$ — effect concentrated in the loss region
\end{itemize}}
\end{block}

\column{0.43\textwidth}
\begin{block}{Identification}
{\footnotesize
\begin{itemize}
  \item \textbf{Hypothesis:} $\delta^{\mathrm{MD}}_1 > 0$ — high margin debt attenuates outflows from bottom performers (sticky losers)
  \item \textbf{Mechanism:} Leveraged investors incur financing costs and accept leverage risk in support of their conviction; sunk-cost bias (Thaler, 1980) and loss aversion (Kahneman \& Tversky, 1979) make exit disproportionately aversive
  \item \textbf{Differentiation from H1:} H1 operates at the \emph{top} (amplified inflows); H2 at the \emph{bottom} (suppressed outflows). Symmetric sentiment predicts equal modulation at both ends — asymmetry is the test
  \item \textbf{Exogeneity:} $\Delta\mathrm{MD}_t$ is aggregate and demand-side — a single fund's performance cannot drive FINRA margin balances
\end{itemize}}
\end{block}
\end{columns}
\end{frame}

% ══════════════════════════════════════════════════════════
%  H3 — LOTTERY DEMAND
% ══════════════════════════════════════════════════════════
\begin{frame}{H3 — Lottery Demand: Model}
\begin{columns}[T]
\column{0.54\textwidth}
\begin{block}{The model (Kumar, 2009; Amihud \& Goyenko, 2013)}
{\footnotesize Piecewise base augmented with activeness proxy interaction:}
{\footnotesize
\begin{align*}
\mathrm{Flow}_{i,t} = \;&\alpha_i + \beta_1 R^{\mathrm{LOW}}_{i,t-1} + \beta_2 R^{\mathrm{MID}}_{i,t-1} + \beta_3 R^{\mathrm{HIGH}}_{i,t-1} \\
&+ \delta_1 (R^{\mathrm{LOW}}_{i,t-1} \times D^{\mathrm{SENT}}_t) + \delta_2(\cdots) + \delta_3(\cdots) \\
&+ \phi\,(A_{i,t-1} \times D^{\mathrm{SENT}}_t) \\
&+ \eta\,A_{i,t-1} + \lambda D^{\mathrm{SENT}}_t + \gamma \mathbf{X}_{i,t-1} + \varepsilon_{i,t}
\end{align*}}
{\footnotesize
\begin{itemize}
  \item $A_{i,t-1} = 1 - R^2_{i,t-1}$: activeness proxy from rolling 36-month Carhart regression (Amihud \& Goyenko, 2013); high $A \Rightarrow$ idiosyncratic, benchmark-deviating fund
  \item Robustness: 36-month rolling return skewness as second proxy (Kumar, 2009)
  \item $\eta$: main effect of activeness; $\phi$: sentiment-conditioned amplification
\end{itemize}}
\end{block}

\column{0.43\textwidth}
\begin{block}{Identification}
{\footnotesize
\begin{itemize}
  \item \textbf{Hypothesis:} $\phi > 0$ — during high-sentiment regimes, flows move disproportionately into high-distinctiveness funds \emph{beyond} the performance rank effect
  \item \textbf{Interpretation:} Investors buy the \emph{product attribute} (skewness profile) rather than the track record — delegated sensation seeking
  \item \textbf{Novelty:} Kumar (2009) documents lottery preference in direct equity ownership; Grinblatt \& Keloharju (2009) link sensation seeking to individual trading. This is the first test in the \emph{delegated} context at fund level
\end{itemize}}
\end{block}

\begin{block}{Shadow Price — Preview}
{\footnotesize The $\phi$ and $\eta$ coefficients directly yield the time-varying MRS between alpha and distinctiveness — the psychological premium quantified in H4.}
\end{block}
\end{columns}
\end{frame}

% ══════════════════════════════════════════════════════════
%  H4 — FEE ELASTICITY / PSYCHOLOGICAL PREMIUM
% ══════════════════════════════════════════════════════════
\begin{frame}{H4 — Fee Elasticity \& the Shadow Price}
\begin{columns}[T]
\column{0.54\textwidth}
\begin{block}{The model (Statman, 2004; Berk \& Green, 2004)}
{\footnotesize Full specification adds expense ratio interaction:}
{\footnotesize
\begin{align*}
\mathrm{Flow}_{i,t} = \;&\alpha_i + \beta_1 R^{\mathrm{LOW}}_{i,t-1} + \cdots + \beta_3 R^{\mathrm{HIGH}}_{i,t-1} \\
&+ \delta_1(\cdots) + \delta_2(\cdots) + \delta_3(\cdots) \\
&+ \phi\,(A_{i,t-1} \times D^{\mathrm{SENT}}_t) + \eta\,A_{i,t-1} \\
&+ \psi\,(\mathrm{ER}_{i,t-1} \times D^{\mathrm{SENT}}_t) \\
&+ \xi\,\mathrm{ER}_{i,t-1} + \lambda D^{\mathrm{SENT}}_t + \gamma \mathbf{X}_{i,t-1} + \varepsilon_{i,t}
\end{align*}}
{\footnotesize
\begin{itemize}
  \item $\mathrm{ER}_{i,t-1}$: expense ratio; $\xi < 0$ expected (baseline fee aversion)
  \item $\psi > 0$: fee aversion weakens under high sentiment — investors treat active management as a consumption good
\end{itemize}}
\end{block}

\column{0.43\textwidth}
\begin{block}{Identification}
{\footnotesize
\begin{itemize}
  \item \textbf{H4 test:} $\psi > 0$ — fee elasticity attenuated in high-sentiment regimes
  \item \textbf{Shadow Price test:} $\mathrm{SP}^{\mathrm{HIGH}} < \mathrm{SP}^{\mathrm{LOW}} < 0$ — investors sacrifice more alpha per unit of distinctiveness when optimistic
  \item \textbf{Example:} $\mathrm{SP} = -0.20$ implies investors treat a 1-unit increase in $1-R^2$ as equivalent to 20 bps of alpha
\end{itemize}}
\end{block}

\begin{block}{Shadow Price (Psychological Premium)}
{\footnotesize From $d\,\mathrm{Flow} = 0$:
\[ \mathrm{SP}_t = -\tfrac{\eta + \phi \cdot D^{\mathrm{HIGH}}_t}{\beta_{\mathrm{rank}} + \delta_{\mathrm{rank}} \cdot D^{\mathrm{HIGH}}_t} \]
Denominated in \textbf{bps of sacrificed alpha per unit of distinctiveness}.}
\end{block}

\begin{block}{Structural contribution}
{\footnotesize Statman (2004) argues \emph{conceptually} that investors pay for psychological utility. This model puts a \textbf{dimensioned, welfare-relevant number} on it — the move from sign prediction to structural quantification is the central original contribution.}
\end{block}

\end{columns}
\end{frame}
